ve_golden_age = {
	scope = country
	is_valid = {
		custom_tooltip = {
			text = ve_golden_age_tt
			NOT = {
				has_variable = clicked_golden_age
			}
			golden_age_3_obj_trigger = yes
		}
	}
	effect = {
		set_variable = clicked_golden_age
		add_modifier = {
			name = golden_age_mdf
			years = 20
		}
	}
}

ve_age_obj_1 = {
	scope = country
	is_valid = {
		trigger_if = {
			limit = {
				has_global_variable = ve_age_1_started
			}
			root.average_sol >= 10
		}
		trigger_else_if = {
			limit = {
				has_global_variable = ve_age_2_started
			}
			hidden_trigger = {
				OR = {
					has_law = law_type:law_colonial_resettlement
					has_law = law_type:law_colonial_exploitation
				}
			}
		}
		trigger_else_if = {
			limit = {
				has_global_variable = ve_age_3_started
			}
			hidden_trigger = {
				NOT = {
					has_law = law_type:law_monarchy
				}
			}
		}
	}
	effect = {
		custom_tooltip = {
			text = ve_age_obj_efct_tt
		}
	}
}

ve_age_bonus_1 = {
	scope = country
	is_shown = {
		trigger_if = {
			limit = {
				has_global_variable = ve_age_1_started
			}
			NOT = {
				has_modifier = age_1_bonus_1
			}
		}
		trigger_else_if = {
			limit = {
				has_global_variable = ve_age_2_started
			}
			NOT = {
				has_modifier = age_2_bonus_1
			}
		}
		trigger_else_if = {
			limit = {
				has_global_variable = ve_age_3_started
			}
			NOT = {
				has_modifier = age_3_bonus_1
			}
		}
	}
	is_valid = {
		trigger_if = {
			limit = {
				has_global_variable = ve_age_1_started
			}
			custom_tooltip = {
				text = splendor_default_bonus
				var:ve_splender_points >= 800
			}
		}
		trigger_else_if = {
			limit = {
				has_global_variable = ve_age_2_started
			}
			custom_tooltip = {
				text = splendor_default_bonus
				var:ve_splender_points >= 800
			}
		}
		trigger_else_if = {
			limit = {
				has_global_variable = ve_age_3_started
			}
			custom_tooltip = {
				text = splendor_default_bonus
				var:ve_splender_points >= 800
			}
		}
	}
	effect = {
		change_variable = {
			name = ve_splender_points
			subtract = 800
		}
		if = {
			limit = {
				has_global_variable = ve_age_1_started
			}
			add_modifier = {
				name = age_1_bonus_1
				months = -1
			}
		}
		else_if = {
			limit = {
				has_global_variable = ve_age_2_started
			}
			add_modifier = {
				name = age_2_bonus_1
				months = -1
			}
		}
		else_if = {
			limit = {
				has_global_variable = ve_age_3_started
			}
			add_modifier = {
				name = age_3_bonus_1
				months = -1
			}
		}
	}
}
