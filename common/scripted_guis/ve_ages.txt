ve_golden_age = {
	scope = country
	is_valid = {
		NOT = {
			has_variable = clicked_golden_age
		}
		# var:ve_triggered_objectives >= 3
		always = yes
	}
	effect = {
		set_variable = clicked_golden_age
	}
	ai_chance = {
		base = 100
	}
}

ve_age_bonus_1 = {
	scope = country
	is_valid = {
		any_country = {
			is_ai = no
			var:ve_splender_points >= 1
			trigger_if = {
				limit = {
					has_variable = ve_age_1_started
				}
				always = yes
			}
			trigger_else_if = {
				limit = {
					has_variable = ve_age_2_started
				}
				always = yes
			}
			trigger_else_if = {
				limit = {
					has_variable = ve_age_3_started
				}
				always = yes
			}
		}
	}
	effect = {
		every_country = {
			limit = {
				is_ai = no
			}
			change_variable = {
				name = ve_splender_points
				subtract = 1
			}
			if = {
				limit = {
					has_variable = ve_age_1_started
				}
				#Some modifiers
			}
			else_if = {
				limit = {
					has_variable = ve_age_2_started
				}
				#Some modifiers
			}
			else_if = {
				limit = {
					has_variable = ve_age_3_started
				}
				#Some modifiers
			}
		}
	}
}
