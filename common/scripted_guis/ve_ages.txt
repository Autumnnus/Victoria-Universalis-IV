ve_age_obj_1 = {
	scope = country
	is_valid = {
		trigger_if = {
			limit = {
				has_variable = ve_age_1_started
			}
			root.average_sol >= 10
		}
		trigger_else_if = {
			limit = {
				has_variable = ve_age_2_started
			}
			OR = {
				has_law = law_type:law_colonial_resettlement
				has_law = law_type:law_colonial_exploitation
			}
		}
		trigger_else_if = {
			limit = {
				has_variable = ve_age_3_started
			}
			NOT = {
				has_law = law_type:law_monarchy
			}
		}
	}
}

### Age Bonuses
ve_age_bonus_1 = {
	scope = country
	is_shown = {
		NOT = {
			OR = {
				has_modifier = age_1_bonus_1
				has_modifier = age_2_bonus_1
				has_modifier = age_3_bonus_1
			}
		}
	}
	is_valid = {
		trigger_if = {
			limit = {
				has_variable = ve_age_1_started
			}
			var:ve_splender_points >= 800
			NOT = {
				has_modifier = age_1_bonus_1
			}
		}
		trigger_else_if = {
			limit = {
				has_variable = ve_age_2_started
			}
			var:ve_splender_points >= 800
			NOT = {
				has_modifier = age_1_bonus_2
			}
		}
		trigger_else_if = {
			limit = {
				has_variable = ve_age_3_started
			}
			var:ve_splender_points >= 800
			NOT = {
				has_modifier = age_1_bonus_3
			}
		}
	}
	effect = {
		change_variable = {
			name = ve_splender_points
			subtract = 800
		}
		if = {
			limit = {
				has_variable = ve_age_1_started
			}
			add_modifier = {
				name = age_1_bonus_1
				months = -1
			}
		}
		else_if = {
			limit = {
				has_variable = ve_age_2_started
			}
			add_modifier = {
				name = age_2_bonus_1
				months = -1
			}
		}
		else_if = {
			limit = {
				has_variable = ve_age_3_started
			}
			add_modifier = {
				name = age_3_bonus_1
				months = -1
			}
		}
	}
}
