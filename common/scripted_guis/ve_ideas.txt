ve_idea_1_embraced_idea = {
	scope = country
	is_valid = {
		any_country = {
			is_ai = no
			NOT = {
				has_variable = idea_1_embraced
			}
			OR = {
				var:ve_max_idea_slot_bar < 8
				NOT = {
					has_variable = ve_max_idea_slot_bar
				}
			}
		}
	}
	effect = {
		every_country = {
			limit = {
				is_ai = no
			}
			if = {
				limit = {
					NOT = {
						has_variable = idea_1_embraced
					}
				}
				set_variable = idea_1_embraced
			}
			set_variable = {
				name = idea_1_bar_point
				value = 0
			}
			if = {
				limit = {
					NOT = {
						has_variable = national_idea_pool
					}
				}
				set_variable = {
					name = national_idea_pool
					value = 0
				}
			}
			if = {
				limit = {
					NOT = {
						has_variable = ve_max_idea_slot_bar
					}
				}
				set_variable = {
					name = ve_max_idea_slot_bar

				}
			}

				}
				change_variable = {
					name = ve_max_idea_slot_bar

		}
	}
}

boost_idea_group = {
	scope = country
	# saved_scopes = {
	# 	idea_group
	# }
	is_shown = {
		switch = {
			trigger = scope:idea_group
			flag:innovative_idea = {
				has_variable = can_go_innovative
				NOT = {
					has_variable = innovative_idea_completed
				}
			}
			flag:economic_idea = {
				has_variable = can_go_economic
			}
		}
	}
	is_valid = {
		switch = {
			trigger = scope:idea_group
			flag:innovative_idea = {
				var:idea_point_pool >= 300
				has_variable = can_go_innovative
			}
		}
	}
	effect = {
		switch = {
			trigger = scope:idea_group
			flag:innovative_idea = {
				if = {
					limit = {
						var:innovative_idea == 0
					}
					change_variable = {
						name = innovative_idea
						add = 1
					}
				}
				else_if = {
					limit = {
						var:innovative_idea == 1
					}
					change_variable = {
						name = innovative_idea
						add = 1
					}
				}
				else_if = {
					limit = {
						var:innovative_idea == 2
					}
					change_variable = {
						name = innovative_idea
						add = 1
					}
				}
				else_if = {
					limit = {
						var:innovative_idea == 3
					}
					change_variable = {
						name = innovative_idea
						add = 1
					}
				}
				else_if = {
					limit = {
						var:innovative_idea == 4
					}
					change_variable = {
						name = innovative_idea
						add = 1
					}
				}
				else_if = {
					limit = {
						var:innovative_idea == 5
					}
					change_variable = {
						name = innovative_idea
						add = 1
					}
				}
				else_if = {
					limit = {
						var:innovative_idea == 6
					}
					change_variable = {
						name = innovative_idea
						add = 1
					}
					set_variable = innovative_idea_completed
				}
			}
		}
	}
}
