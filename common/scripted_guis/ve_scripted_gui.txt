ve_idea_1_embraced_idea = {
	scope = country
	is_valid = {
		any_country = {
			is_ai = no
			NOT = {
				has_variable = idea_1_embraced
			}
		}
	}
	effect = {
		every_country = {
			limit = {
				is_ai = no
			}
			if = {
				limit = {
					NOT = {
						has_variable = idea_1_embraced
					}
				}
				set_variable = idea_1_embraced
			}
			set_variable = {
				name = idea_1_bar_point
				value = 0
			}
			if = {
				limit = {
					NOT = {
						has_variable = national_idea_pool
					}
				}
				set_variable = {
					name = national_idea_pool
					value = 0
				}
			}
		}
	}
}

					name = idea_1_bar_point
					add = 10
				}

					name = idea_point_pool
					subtract = 300
				}
				if = {
					limit = {
						var:national_idea_pool < 21
					}
					change_variable = {
						name = national_idea_pool
						add = 1
					}
				}
			}
		}
	}
}

			is_ai = no

			NOT = {

			}
		}

	effect = {
		every_country = {
			limit = {
				is_ai = no
			}
			if = {
				limit = {
					NOT = {
						has_variable = idea_8_but_5_completed
					}
				}
				set_variable = idea_8_but_5_completed
				change_variable = {
					name = idea_8_bar_point
					add = 10
				}
				change_variable = {
					name = idea_point_pool
					subtract = 300
				}
				if = {
					limit = {
						var:national_idea_pool < 21
					}
					change_variable = {
						name = national_idea_pool
						add = 1
					}
				}
			}
		}
	}
}
