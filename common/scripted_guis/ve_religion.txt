ve_holy_site_1 = {
	scope = country
	is_shown = {
		any_country = {
			is_ai = no
			trigger_if = {
				limit = {
					religion = rel:catholic
				}
				owns_entire_state_region = STATE_LAZIO
			}
			trigger_else_if = {
				limit = {
					religion = rel:sunni
				}
				owns_entire_state_region = STATE_HEDJAZ
			}
		}
	}
	effect = {
		every_country = {
			limit = {
				is_ai = no
			}
		}
	}
}
