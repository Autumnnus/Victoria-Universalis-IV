balance_missionaries = {
	if = {
		limit = {
			avaible_missionaries < 0
		}
		every_scope_state = {
			limit = {
				has_modifier = ve_missionary_mdf
			}
			remove_modifier = ve_missionary_mdf
			remove_variable = selected_state_religion
		}
		set_variable = {
			name = avaible_missionaries
			value = total_missionaries
		}
	}
}
