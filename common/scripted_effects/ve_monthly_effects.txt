ve_monthly_points = {
	change_variable = {
		name = idea_point_pool
		add = 14		#1680 for 10 years. For a full idea needs 2100 points
	}
	if = {
		limit = {
			literacy_rate <= 0.25
		}
		change_variable = {
			name = idea_point_pool
			add = 1
		}
	}
	else_if = {
		limit = {
			literacy_rate <= 0.5
		}
		change_variable = {
			name = idea_point_pool
			add = 2
		}
	}
	else_if = {
		limit = {
			literacy_rate <= 0.75
		}
		change_variable = {
			name = idea_point_pool
			add = 3
		}
	}
	else_if = {
		limit = {
			literacy_rate <= 1
		}
		change_variable = {
			name = idea_point_pool
			add = 4
		}
	}
	change_variable = {
		name = piety_bar_point
		add = 4		# 240 for 5 years. An aspect cost is 100 for 5 years.
	}
	if = {
		limit = {
			has_law = law_type:law_theocracy
		}
		change_variable = {
			name = piety_bar_point
			add = 3
		}
	}
	if = {
		limit = {
			has_law = law_type:law_state_religion
		}
		change_variable = {
			name = piety_bar_point
			add = 4
		}
	}
	else_if = {
		limit = {
			has_law = law_type:law_freedom_of_conscience
		}
		change_variable = {
			name = piety_bar_point
			add = 2
		}
	}
	if = {
		limit = {
			OR = {
				religion = rel:catholic
				religion = rel:protestant
			}
			owns_entire_state_region = STATE_NORTH_RHINE
		}
		change_variable = {
			name = piety_bar_point
			add = 2
		}
	}
	else_if = {
		limit = {
			OR = {
				religion = rel:sunni
				religion = rel:shiite
			}
			owns_entire_state_region = STATE_HEDJAZ
		}
		change_variable = {
			name = piety_bar_point
			add = 2
		}
	}
	else_if = {
		limit = {
			religion = rel:hindu
			owns_entire_state_region = STATE_AWADH
		}
		change_variable = {
			name = piety_bar_point
			add = 2
		}
	}
	else_if = {
		limit = {
			religion = rel:sikh
			owns_entire_state_region = STATE_MYSORE
		}
		change_variable = {
			name = piety_bar_point
			add = 2
		}
	}
	if = {
		limit = {
			NOT = {
				var:selected_reform_value == 7
			}
		}
		change_variable = {
			name = culture_reform_point
			add = 6			# 720 for 10 years. A reform cost is 750 and total Reform Cost is 5250
		}
	}
}
