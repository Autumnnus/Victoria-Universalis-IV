# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law
# cultural_acceptance_rule and religious_acceptance_rule are triggers called in culture scope with scope:country as the country with the laws
law_ethnostate = {
	group = lawgroup_citizenship
	icon = "gfx/interface/icons/law_icons/ethnostate.dds"
	progressiveness = -100
	on_enact = {
		recalculate_pop_ig_support = yes
		if = {
			limit = {
				has_variable = ve_had_national_supremacy_law
			}
			change_variable = {
				name = avaible_culture_missionaries
				add = 1
			}
			change_variable = {
				name = total_culture_missionaries
				add = 1
			}
			remove_variable = ve_had_national_supremacy_law
			set_variable = ve_had_ethnostate_law
			balance_cultural_missionaries = yes
		}
		else_if = {
			limit = {
				has_variable = ve_had_racial_segregation_law
			}
			change_variable = {
				name = avaible_culture_missionaries
				add = 2
			}
			change_variable = {
				name = total_culture_missionaries
				add = 2
			}
			remove_variable = ve_had_racial_segregation_law
			set_variable = ve_had_ethnostate_law
			balance_cultural_missionaries = yes
		}
		else_if = {
			limit = {
				has_variable = ve_had_cultural_exclusion_law
			}
			change_variable = {
				name = avaible_culture_missionaries
				add = 3
			}
			change_variable = {
				name = total_culture_missionaries
				add = 3
			}
			remove_variable = ve_had_cultural_exclusion_law
			set_variable = ve_had_ethnostate_law
			balance_cultural_missionaries = yes
		}
		else_if = {
			limit = {
				has_variable = ve_had_multicultural_law
			}
			change_variable = {
				name = avaible_culture_missionaries
				add = 4
			}
			change_variable = {
				name = total_culture_missionaries
				add = 4
			}
			remove_variable = ve_had_multicultural_law
			set_variable = ve_had_ethnostate_law
			balance_cultural_missionaries = yes
		}
	}
	cultural_acceptance_rule = {
		is_primary_culture_of = scope:country
	}
	unlocking_technologies = {
		nationalism
	}
	modifier = {
		country_authority_add = 200
		state_radicals_from_sol_change_accepted_culture_mult = -0.2
		state_loyalists_from_sol_change_accepted_culture_mult = 0.2
	}
	possible_political_movements = {
		law_racial_segregation
		law_cultural_exclusion
	}
	pop_support = {
		value = 0
	}
}
